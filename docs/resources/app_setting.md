# App Setting Resource Example

# Basic App Setting configuration

  name      = "example-app-setting"
  namespace = "staging"

  labels = {
    environment = "production"
    managed_by  = "terraform"
  }

  annotations = {
    "owner" = "platform-team"
  }

  # Resource-specific configuration
