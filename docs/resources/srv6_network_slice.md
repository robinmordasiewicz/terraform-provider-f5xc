# f5xc_srv6_network_slice (Resource)

Manages srv6_network_slice creates a new object in the storage backend for metadata.namespace. in F5 Distributed Cloud.

# Manages srv6_network_slice creates a new object in the storage backend for metadata.namespace. in F5 Distributed Cloud.

  name      = "example-srv6-network-slice"
